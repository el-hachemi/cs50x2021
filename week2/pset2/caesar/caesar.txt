{
int n;

{
  printf("Error: Enter an argument after ./caeser\n");
  return 1; 
}
else if (argc > 2)
{
  printf("Error: Only one argument is required\n");
  return 1;
}
else
{
  for (int i = 0; i != 'NUL'; i++)
  {
    if (k[i] < 48 || k[i] > 57)
    {
      printf("Usage: ./caesar key\n");

    }
  }

    c[j] = (p[j] + k) % 26;
